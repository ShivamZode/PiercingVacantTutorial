from datetime import datetime

# Assuming obj1 is an instance of your model with DateTimeField
# and datetime_str is the datetime string
datetime_str = "2024-05-12 15:30:00"

# Convert the string to a datetime object
datetime_obj2 = datetime.strptime(datetime_str, "%Y-%m-%d %H:%M:%S")

# Subtract the datetime objects
time_difference = obj1.created_at - datetime_obj2

# Convert the time difference to seconds
time_difference_in_seconds = time_difference.total_seconds()
